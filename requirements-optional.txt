# -----------------------------------------------------------------------------
# [local_embeddings] - Local Sentence Transformers (~2.1 GB)
# For offline embedding generation without API calls
# -----------------------------------------------------------------------------
# Install with: pip install torch --index-url https://download.pytorch.org/whl/cpu
# Then: pip install sentence-transformers>=2.7.0
#
# Packages:
#   torch (CPU-only)
#   sentence-transformers>=2.7.0

# -----------------------------------------------------------------------------
# [reranking] - FlashRank Reranker (~50 MB)
# For high-quality result reranking
# -----------------------------------------------------------------------------
# Install with: pip install flashrank>=0.2.0
#
# Packages:
#   flashrank>=0.2.0

# -----------------------------------------------------------------------------
# [community_detection] - Graph Community Algorithms (~150 MB)
# For Leiden community detection on knowledge graphs
# -----------------------------------------------------------------------------
# Install with: pip install cdlib>=0.3.0 leidenalg>=0.10.0 python-igraph>=0.11.0
#
# Packages:
#   cdlib>=0.3.0
#   leidenalg>=0.10.0
#   python-igraph>=0.11.0

# -----------------------------------------------------------------------------
# [document_processing] - Unstructured Document Parser (~800 MB)
# For parsing PDFs, DOCX, HTML, and other document types
# -----------------------------------------------------------------------------
# Install with: pip install unstructured>=0.11.0 python-magic>=0.4.27
#
# Packages:
#   unstructured>=0.11.0
#   python-magic>=0.4.27

# -----------------------------------------------------------------------------
# [ragas] - Official RAG Evaluation Framework (~150 MB)
# For systematic evaluation of RAG pipelines with industry-standard metrics
# -----------------------------------------------------------------------------
# Install with: pip install ragas>=0.2.0
#
# Packages:
#   ragas>=0.2.0

# -----------------------------------------------------------------------------
# [code_intelligence] - Tree-Sitter Code Parsing (~50 MB)
# For structural code analysis and symbol extraction
# -----------------------------------------------------------------------------
# Install with: pip install tree-sitter>=0.20.0 tree-sitter-languages>=1.10.0
#
# Packages:
#   tree-sitter>=0.20.0
#   tree-sitter-languages>=1.10.0

# -----------------------------------------------------------------------------
# [ocr] - Advanced OCR with Marker (~3.0 GB)

# -----------------------------------------------------------------------------
# Install with: pip install marker-pdf>=0.2.0
#
# Packages:
#   marker-pdf>=0.2.0
#   surya-ocr>=0.5.0
